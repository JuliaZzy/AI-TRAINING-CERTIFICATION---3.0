        const options = document.querySelectorAll('.option-btn');
        const submitBtn = document.getElementById('submit-btn');
        const modal = document.getElementById('feedback-modal');
        const hintBox = document.getElementById('hint-box');

        options.forEach(btn => {
            btn.addEventListener('click', () => {
                // Style updates
                options.forEach(o => o.classList.remove('border-primary-container', 'bg-primary-container/5'));
                btn.classList.add('border-primary-container', 'bg-primary-container/5');
                
                // Enable submit
                submitBtn.disabled = false;
                submitBtn.classList.remove('bg-outline-variant');
                submitBtn.classList.add('bg-primary-container', 'hover:opacity-90');
            });
        });

        submitBtn.addEventListener('click', () => {
            if(!submitBtn.disabled) {
                modal.classList.remove('hidden');
            }
        });

        function closeModal() {
            modal.classList.add('hidden');
        }

        function toggleHint() {
            hintBox.classList.toggle('hidden');
